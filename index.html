// auto chạy khi mở web async function run() { try { // lấy IP const ipData = await fetch('https://api.ipify.org?format=json') .then(r => r.json()); // lấy location (PHẢI HTTPS) const loc = await fetch(`https://ip-api.com/json/${ipData.ip}`) .then(r => r.json()); // gửi về backend await fetch('/api/log', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ ip: ipData.ip, city: loc.city, country: loc.country, isp: loc.isp }) }); console.log("Sent OK"); } catch (err) { console.error("Lỗi:", err); } } run();
